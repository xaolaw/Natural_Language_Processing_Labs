import re

# Lista miesicy w rnych odmienionych formach
months = [
    "stycze", "stycznia", "luty", "lutego", "marzec", "marca",
    "kwiecie", "kwietnia", "maj", "maja", "czerwiec", "czerwca",
    "lipiec", "lipca", "sierpie", "sierpnia", "wrzesie", "wrzenia",
    "padziernik", "padziernika", "listopad", "listopada", "grudzie", "grudnia"
]

# Tworzenie regex na podstawie miesicy
months_pattern = '|'.join(months)

# Regex do znajdowania dat
date_pattern = rf"(\b(\d{{1,2}} )?({months_pattern}) (\d{{4}}|\d{{2}})?\b|\b({months_pattern}) (\d{{1,2}})\b)"

# Testowy tekst
text = "20 padziernik, listopad 2012, luty 21, maj 16, stycze 21, czerwiec 10, sierpie 1999, 10 wrzenia 2019."

# Znajdowanie dat
matches = re.findall(date_pattern, text)

# Wyodrbnianie wynikw
found_dates = [match[0] for match in matches]
